JavaScript

hx.select('#example')
  .add(hx.checkbox())
  .add(hx.button({context: 'positive'}).text('Positive Button'))
  .add(hx.button({context: 'warning'}).text('Warning Button'))
  .add(hx.button({context: 'action'}).text('Action Button'))
  .add(hx.icon({class: 'fa fa-plus'}))
