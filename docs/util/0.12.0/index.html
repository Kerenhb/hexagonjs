  { type: 'b', value: 5 },

  [
    [
      "a", [
        { type: 'a', value: 1 },
        { type: 'a', value: 2 }

    [
      "b", [
        { type: 'b', value: 3 },
        { type: 'b', value: 4 },
        { type: 'b', value: 5 }
      ]
    ],
    [
      "c", [
        { type: 'c', value: 6 }
      ]
    ]
  ]

  return v.type;
})

Returns

hx.mergeretainUndefinedBooleanobjects...ObjectObject

objects...

Returns
